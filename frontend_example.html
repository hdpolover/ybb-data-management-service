getExportData();
    
    // Configure export
    $exportData = [
        'data' => $participants,
        'template' => $this->request->getPost('template') ?: 'standard',
        'format' => 'excel',
        'filename' => $this->request->getPost('filename') ?: null,
        'sheet_name' => $this->request->getPost('sheet_name') ?: 'Participants Data'
    ];
    
    // Call Python API
    $apiUrl = 'http://127.0.0.1:5000/api/ybb/export/participants';
    
    $ch = curl_init();
    curl_setopt_array($ch, [
        CURLOPT_URL => $apiUrl,
        CURLOPT_POST => true,
        CURLOPT_POSTFIELDS => json_encode($exportData),
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_TIMEOUT => 300,
        CURLOPT_HTTPHEADER => [
            'Content-Type: application/json',
            'Accept: application/json'
        ]
    ]);
    
    $response = curl_exec($ch);
    $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);
    
    if ($httpCode === 200) {
        $result = json_decode($response, true);
        
        if ($result['status'] === 'success') {
            // Return download information to frontend
            return $this->response->setJSON([
                'success' => true,
                'export_id' => $result['data']['export_id'],
                'filename' => $result['data']['file_name'],
                'download_url' => base_url("admin/download/{$result['data']['export_id']}"),
                'file_size' => $result['data']['file_size'],
                'record_count' => $result['data']['record_count']
            ]);
        }
    }
    
    return $this->response->setJSON([
        'success' => false,
        'message' => 'Export failed'
    ]);
}

// Download handler
public function download($exportId)
{
    $downloadUrl = "http://127.0.0.1:5000/api/ybb/export/{$exportId}/download";
    
    // Stream file from Python API to browser
    $ch = curl_init();
    curl_setopt_array($ch, [
        CURLOPT_URL => $downloadUrl,
        CURLOPT_RETURNTRANSFER => true,
        CURLOPT_TIMEOUT => 120
    ]);
    
    $fileContent = curl_exec($ch);
    $httpCode = curl_getinfo($ch, CURLINFO_HTTP_CODE);
    curl_close($ch);
    
    if ($httpCode === 200) {
        return $this->response
            ->setHeader('Content-Type', 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet')
            ->setHeader('Content-Disposition', 'attachment; filename="participants_export.xlsx"')
            ->setBody($fileContent);
    }
    
    throw new \CodeIgniter\Exceptions\PageNotFoundException();
}
